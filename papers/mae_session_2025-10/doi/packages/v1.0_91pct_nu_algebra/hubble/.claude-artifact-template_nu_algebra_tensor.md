## 11. CODE STRUCTURE (JavaScript/React)

```javascript
// Calculate epistemic distance
function calculateEpistemicDistance(T1, T2) {
  const dP = T1[0] - T2[0];
  const dt = T1[1] - T2[1];
  const dm = T1[2] - T2[2];
  const da = T1[3] - T2[3];

  return Math.sqrt(dP*dP + dt*dt + dm*dm + da*da);
}

// Tensor-extended merge
function tensorMerge(early, late, deltaT) {
  const n_merged = (early.H0 + late.H0) / 2;
  const disagreement = Math.abs(early.H0 - late.H0);

  // Standard uncertainty
  const u_std = Math.sqrt((early.sigma**2 + late.sigma**2) / 2);

  // Tensor expansion
  const tensor_expansion = (disagreement / 2) * deltaT;
  const u_total = u_std + tensor_expansion;

  return {
    H0: n_merged,
    sigma: u_total,
    interval: [n_merged - u_total, n_merged + u_total],
    expansion_ratio: u_total / u_std
  };
}

// Check concordance
function checkConcordance(merged, early, late) {
  const early_contained =
    early.interval[0] >= merged.interval[0] &&
    early.interval[1] <= merged.interval[1];

  const late_contained =
    late.interval[0] >= merged.interval[0] &&
    late.interval[1] <= merged.interval[1];

  let gap = 0;
  if (!late_contained) {
    gap = Math.max(0, late.interval[0] - merged.interval[1]);
  }

  return { early_contained, late_contained, gap };
}
```
